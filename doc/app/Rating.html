def after_update
  if self.grade_changed?
    before_grade, after_grade = self.grade_change
    self.manifesto.decrement_with_sql!(GRADE_MAP[before_grade])
    self.manifesto.increment_with_sql!(GRADE_MAP[after_grade])
  end
end
